<script>

//defining a function to validate form

//const frm=document.forms[0];
const frm=document.querySelector("#f1");

frm.addEventListener("submit", checkData );
 
function checkData(f)
{
  //fullname checking
  const fn=document.querySelector("#fna").value;
  const fnexp=/^[a-zA-Z ]*$/ 
  
  if( fnexp.test(fn)===false || fn==="")
  {
	document.querySelector("#errfna").textContent=`plz enter a valid name`;
	f.preventDefault();
  }else
	document.querySelector("#errfna").textContent=``;
	
  //phone checking
  const p=document.querySelector("#ph").value;
  const phexp=/^[6-9]\d{9}$/;
  
  if(phexp.test(p)===false)
  {
	document.querySelector("#errph").textContent=`plz enter a valid contact number`;
	f.preventDefault();
  }else
    document.querySelector("#errph").textContent=``;
  
  //pincode checking
  const pc=document.querySelector("#pin").value;
  const pcexp=/^\d{6}$/;
  
  if(pcexp.test(pc)===false)
  {
	document.querySelector("#errpin").textContent=`plz enter 6digit code`;
	f.preventDefault();
  }else
    document.querySelector("#errpin").textContent=``;  

  //password checking
  const pw=document.querySelector("#pwd").value;
  const pwexp=/^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.*[@#$%*&]).{8,20}$/;
  
  if(pwexp.test(pw)===false)
  {
	document.querySelector("#errpwd").textContent=`password should contains 1 lower, upper, digit & symbol`;
	f.preventDefault();
  }else
    document.querySelector("#errpwd").textContent=``;  	
} //fun end

</script>
